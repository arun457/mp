.model small
.stack 100h

.data
    array db 3,1,2,5,4  

.code
main proc
    mov ax, @data
    mov ds, ax
     
    lea si, array
    mov ch, 5h
    
out_loop:mov al, [si]
    mov cl, 5h
    lea di, array 
    
in_loop:cmp al, [di]
     jnc skip       ;change to jc for descending
     xchg al, [di]
     mov [si],al
skip:inc di
     dec cl
     jnz in_loop
     
     inc si
     dec ch
     jnz out_loop
     
     
     mov ah, 02h
     mov ch, 05
     lea si, array     
display:mov dl, [si] 
    add dl, 48
    int 21h
    
    mov dl, ' '
    int 21h
    
    inc si
    dec ch
    jnz display
            
    mov ah, 4ch
    int 21h
main endp
end main
