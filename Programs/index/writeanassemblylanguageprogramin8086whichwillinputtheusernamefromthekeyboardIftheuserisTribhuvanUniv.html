                                                     .model small
.stack 100h
.data 
    msg db "Enter a string (length<100) ending with$"   ;optional
    username db "Tribhuwan University"
    valid_msg db "The Username is valid.$"
    invalid_msg db "Username Invalid.$"
    str db 100 dup(0)          ;to store input string
    flag db 00h                ;flag to indicate yes or no
.code
    main proc 
            
        mov ax, @data
        mov ds, ax
                                                             
        newline macro ;newline and carriage return
            push ax
            mov ah, 02h
            mov dl, 10
            int 21h
            mov dl, 0dh
            int 21h
            pop ax
        endm    
        
        
         
        ;-------------Optional prompt to user  
        ;displaying prompt message to user
        lea dx, msg
        mov ah, 09h
        int 21h
        mov ah, 02h
        mov dl, '$'
        int 21h
        
        newline
        ;-------------optional ends  
          
          
            
        ;taking input from the user
        lea bx, str
        mov ah, 01h
input_loop:int 21h
        cmp al, '$'
        mov [bx], al
        je stop_inputs
        inc bx
        jmp input_loop    
          
stop_inputs:  
        newline
        
        ;checking the string with username 
        mov cx, 20     
        lea si, str
        lea di, username
  check:mov dl, [si]
        cmp dl, '$'
        je invalid
        cmp dl, [di]
        jne invalid
  
        inc si
        inc di
        loop check
        
        cmp [si], '$'
        jne invalid
        ;-------- finished checking the string with username
 
        
        lea dx, valid_msg
        jmp display_message
invalid:lea dx, invalid_msg  
        
display_message: mov ah, 09h 
        int 21h        
        
          
                          
      
      mov ah, 4ch
      int 21h
       
   main endp
end main
