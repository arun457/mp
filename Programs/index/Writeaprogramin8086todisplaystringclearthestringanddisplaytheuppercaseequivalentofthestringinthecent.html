.model small
.stack 100h

.data
msg db "hello world$"
x db 10
y db 10
.code
main proc
    mov ax, @data
    mov ds, ax
    
    ; display original string
    mov ah, 09h
    mov dx, offset msg
    int 21h
      
    ;clearing the screen and setting to video mode                                             
    mov ah, 00h
    mov al, 03h
    int 10h
     
    ;moving the cursor to the center of screen 
    mov ah, 02h
    mov bh, 00h
    mov dh, y
    mov dl, x
    int 10h     
         
         
    ; convert string to uppercase
    lea si, msg
convert_loop:
    mov dl, [si]
    cmp dl, '$'
    je convert_done
    cmp dl, 'a'
    jc skip_convert
    sub dl, 32 
    
skip_convert:
    int 21h
    inc si
    jmp convert_loop
    
    
convert_done: 
    mov ah, 4ch
    int 21h
main endp
end main
